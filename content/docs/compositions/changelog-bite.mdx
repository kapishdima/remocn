## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { ChangelogBite } from "@/components/remocn/changelog-bite";

const Bite = () => (
  <ChangelogBite
    label="New"
    title="Inline diff view"
    accent="#FFB38E"
  />
);

export const RemotionRoot = () => (
  <Composition
    id="ChangelogBite"
    component={Bite}
    durationInFrames={150}
    fps={30}
    width={1080}
    height={1080}
  />
);
```
