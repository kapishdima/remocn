## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { SuccessConfetti } from "@/components/remocn/success-confetti";

const SuccessConfettiScene = () => (
  <SuccessConfetti
    text="Merged!"
    count={60}
    colors={["#ff5e3a", "#22c55e", "#0ea5e9", "#facc15", "#a855f7"]}
  />
);

export const RemotionRoot = () => (
  <Composition
    id="SuccessConfetti"
    component={SuccessConfettiScene}
    durationInFrames={120}
    fps={30}
    width={1280}
    height={720}
  />
);
```
