## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { GridPixelateWipe } from "@/components/remocn/grid-pixelate-wipe";

const GridPixelateWipeScene = () => (
  <GridPixelateWipe
    from={<SceneA />
);

export const RemotionRoot = () => (
  <Composition
    id="GridPixelateWipe"
    component={GridPixelateWipeScene}
    durationInFrames={90}
    fps={30}
    width={1280}
    height={720}
  />
);
```
