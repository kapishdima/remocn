## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { FrostedGlassWipe } from "@/components/remocn/frosted-glass-wipe";

const FrostedGlassWipeScene = () => (
  <FrostedGlassWipe
    from={<SceneA />
);

export const RemotionRoot = () => (
  <Composition
    id="FrostedGlassWipe"
    component={FrostedGlassWipeScene}
    durationInFrames={90}
    fps={30}
    width={1280}
    height={720}
  />
);
```
