## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { DirectionalWipe } from "@/components/remocn/directional-wipe";

const DirectionalWipeScene = () => (
  <DirectionalWipe
    from={<SceneA />
);

export const RemotionRoot = () => (
  <Composition
    id="DirectionalWipe"
    component={DirectionalWipeScene}
    durationInFrames={90}
    fps={30}
    width={1280}
    height={720}
  />
);
```
